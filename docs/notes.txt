from django import models

from extensible.models import ExtensibleModelMixin


class Customer(ExtensiblelModelMixin, models.Model):

    name = models.CharField()
    age = models.PositiveIntegerField()


# And in return you get the models behave as normal for Django, plus:
#
# - Additionally dynamically generated forms in the admin interface
# - Strong checking of content in these extra fields
# - Querying within these fields
# - Extensions for Django Model Forms (and Crispy Forms)
# - Extensions for Django Rest Framework
# - Extensions for Django (Rest Framework) Filters
# - Migrations
# - ...
#
# Everything just works, but in addition to the core fields that have
# strong guarantees at the DB level.
